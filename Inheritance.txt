using System;namespace inheritance{    class Group    {        public string groupName = "Hikers R Us";        public void greet()        {            Console.WriteLine("Welcome back!");        }    }    class User : Group // User inherites from Group    {        public string userName = "jsmith14";    }    class Program    {        static void Main(string[] args)        {            User newUser = new User();            newUser.greet();            Console.WriteLine(newUser.userName + " is a member of " + newUser.groupName);        }    }}

Oink!");            }        }        class Duck : Animal        {            public override void sound()            {                Console.WriteLine("Quack! Quack!");            }        }                static void Main(string[] args)        {            Animal myAnimal =  new Animal();            Animal myPig = new Pig();            Animal myDuck = new Duck();            myAnimal.sound();            myPig.sound();            myDuck.sound();        }    }}
